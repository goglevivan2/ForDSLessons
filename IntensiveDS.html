def de_mean(x):
    x_bar=mean(x)
    return[x_i-x_bar for x_i in x]
In [45]:
def dot(v,w):
    return sum(v_i*w_i for v_i,w_i in zip(v,w))

def sum_of_squares(v):
    return dot(v,v)

def variance(x):
    n=len(x)
    deviations=de_mean(x)
    return sum_of_squares(deviations)/(n-1)
